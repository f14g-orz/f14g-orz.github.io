从题目分析可以看出这是一个简单NTRU的题目，我们通过构造格来解决这个问题

h = inverse(f, p) * g % p
c = (r*h + m) % p

$ h \equiv f^{-1} g mod p$

$ c \equiv (rh + m ) mod p$

$c \equiv (rf^{-1}g+m) mod p$

$fc \equiv (rg+fm) mod p$

在模上g

$fc \equiv fm mod g mod p$

再乘以inverse(f,g)

$m \equiv fc mod p mod g*inverse(f,g)mod g$

$ h \equiv f^{-1} g mod p$

$g=dh modp$

$g = fh + kp$

​ $ \begin{bmatrix} f & k \end{bmatrix} \begin{bmatrix} 1 & h \ 0 & g \end{bmatrix} \tag{1} = \begin{bmatrix} f & g \end{bmatrix} $

def SmartAttack(P,Q,p):
    E = P.curve()
    Eqp = EllipticCurve(Qp(p, 2), [ ZZ(t) + randint(0,p)*p for t in E.a_invariants() ])

    P_Qps = Eqp.lift_x(ZZ(P.xy()[0]), all=True)
    for P_Qp in P_Qps:
        if GF(p)(P_Qp.xy()[1]) == P.xy()[1]:
            break

    Q_Qps = Eqp.lift_x(ZZ(Q.xy()[0]), all=True)
    for Q_Qp in Q_Qps:
        if GF(p)(Q_Qp.xy()[1]) == Q.xy()[1]:
            break

    p_times_P = p*P_Qp
    p_times_Q = p*Q_Qp

    x_P,y_P = p_times_P.xy()
    x_Q,y_Q = p_times_Q.xy()

    phi_P = -(x_P/y_P)
    phi_Q = -(x_Q/y_Q)
    k = phi_Q/phi_P
    return ZZ(k)

r = SmartAttack(P, Q, p)
print(r)
print(long_to_bytes(r))
